def smooth_pdfs(with_nfci, without_nfci):
    minn, maxx = min(with_nfci.min(), without_nfci.min()), max(with_nfci.max(), without_nfci.max())
    
    with_nfci = st.nct(*st.nct.fit(with_nfci.astype('float64'))).pdf(np.linspace(minn, maxx, 1000))
    without_nfci = st.nct(*st.nct.fit(without_nfci.astype('float64'))).pdf(np.linspace(minn, maxx, 1000))
    output = pd.DataFrame({"NFCI.L1 & GDP.L1":with_nfci, "GDP.L1":without_nfci}, index = np.linspace(minn, maxx, 1000))
    return output

quantiles =[q/500.0 for q in range(1,500)]

WITH NFCI¶

In [18]:
quantile_estimates_with_nfci = pd.DataFrame(columns = subsample_countries, index = range(1,501))
for country in subsample_countries:
    regression_data = pd.DataFrame({"GDP": Y_GDP[country], "GDP_L1": X_GDP_L1[country],"NFCI_L1": X_NFCI_L1[country]})
    quantile_estimates_with_nfci.loc[500, country] = regression_data.tail(1)['GDP'][0]
    
    for q in quantiles:
        mod = smf.quantreg('GDP ~ GDP_L1 + NFCI_L1', regression_data[:-1]).fit(q=q,max_iter=1000)
        quantile_estimates_with_nfci.loc[int(round(q*500)), country] = mod.predict(regression_data[-1:])[0]

fig, axs = plt.subplots(4, 3, figsize = (25, 20))
fig.suptitle("DISTRIBUTION OF CONDITIONAL AND UNCONDITIONAL 1 STEP AHEAD GDP GROWTH FORECAST FOR {}".format(gdp.index.max().strftime('%b %Y')), y = 1)
for i,country in enumerate(subsample_countries):
    smooth_pdfs(quantile_estimates_with_nfci[country][:-1], quantile_estimates_without_nfci[country][:-1]).plot(ax = axs[i/3, i%3], linewidth=2)
    axs[i/3, i%3].axvline(x=quantile_estimates_with_nfci.tail(1)[country].iloc[0], color = 'k', linewidth = 3)
    axs[i/3, i%3].set_title(country)
    axs[i/3, i%3].set(xlabel='Growth Rate', ylabel='Probability')
plt.tight_layout(pad=3.0)
plt.savefig('graphs/Cond_Uncond_forecast_distribution_normal.png')
