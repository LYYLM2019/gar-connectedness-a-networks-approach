def calculate_descriptive_statistic(data, name = False):
    output = pd.DataFrame(columns = data.columns.values, index = ['Mean','Median','Min','25th Quantile','50th Quantile','75th Quantile','Max', 'Standard Deviation','Skewness','Kurtosis','Auto Correlation (1)','Auto Correlation (4)','Auto Correlation (20)'])
    for col in data.columns:
        stats = [
        data[col].mean(),
        data[col].median(),
        data[col].min(),
        data[col].quantile(0.25),
        data[col].quantile(0.50),
        data[col].quantile(0.75),
        data[col].max(),
        data[col].std(),
        data[col].skew(),
        data[col].kurtosis(),
        data[col].corr(data[col].shift(1)),
        data[col].corr(data[col].shift(4)),
        data[col].corr(data[col].shift(20))]
        output[col] = stats
    if name:
        f = open("tables/{}.txt".format(name),"w") 
        output.round(4).to_latex(buf = f, column_format = 'l|r|r|r|r|r|r|r|r|r|r|r|r|r|r') 
        f.close()
        
    return output

def estimate_quantile_forecast(data, name, quantile, exp_win_start = None):
    first_forecast_index = data.index.get_loc(exp_win_start)
    prediction  = pd.DataFrame({name: np.nan}, index = data[first_forecast_index:].index)
    
    for t in range(prediction.shape[0]):
        mod = smf.quantreg('GDP ~ '+" + ".join([c for c in data.columns if '_L' in c]), data[:first_forecast_index+t]).fit(q=quantile,max_iter=2000)
        prediction.iloc[t][name] = mod.predict(data.iloc[first_forecast_index+t])[0]
    return prediction
